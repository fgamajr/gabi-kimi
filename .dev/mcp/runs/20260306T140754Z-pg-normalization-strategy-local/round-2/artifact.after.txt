# Benchmark Normalization Plan: Field Coverage Fix at Adapter Layer

## 0. Executive Summary

This plan addresses benchmark bias where PG underperforms due to missing `issuing_organ` and `body_plain` fields in adapter output. The root cause is adapter-level field projection, not grader defect. This revision addresses all adversarial objections including: cardinality protection for JOINs, TOAST I/O overhead, null frequency parity, ES/PG schema normalization, hybrid merge strategy, and validation performance.

```python
def capture_baseline_latency(adapter, queries, k=100):
    """Capture p50/p95/p99 latency on CURRENT adapter before changes."""
    latencies = []
    for query in queries:
        start = time.perf_counter()
        adapter.search(query, k=k)
        latencies.append((time.perf_counter() - start) * 1000)
    return {
        "p50": sorted(latencies)[len(latencies)//2],
        "p95": sorted(latencies)[int(len(latencies)*0.95)],
        "p99": sorted(latencies)[int(len(latencies)*0.99)],
    }
```

**Decision Matrix:**
- If PG schema lacks field → schema migration required before benchmark
- If PG adapter omits field in SELECT → fix adapter projection
- If PG content ≠ body_plain → use `content` for tsvector AND aliased as `body_plain` for consistency
- If ES index missing field → reindex required before benchmark
- If documents:organizations is 1:N → use DISTINCT ON or aggregate strategy
- If body_plain avg size >1KB → use body_snippet (first 1KB) instead

## 3. Normalization Layer (Base Adapter Class)

```python
# src/adapters/base.py
from abc import ABC, abstractmethod
from dataclasses import dataclass
from typing import Any

@dataclass
class SearchResult:
    id: str
    score: float
    rank: int
    issuing_organ: str | None = None
    body_plain: str | None = None
    body_snippet: str | None = None
    title: str | None = None
    url: str | None = None
    date: str | None = None
    metadata: dict[str, Any] = None
    
    def __post_init__(self):
        """Runtime validation - much faster than JSONSchema."""
        if self.metadata is None:
            self.metadata = {}

class BaseAdapter(ABC):
    @abstractmethod
    def search(self, query: str, k: int = 100) -> list[SearchResult]:
        pass
    
    def _normalize_es_result(self, hit: dict) -> SearchResult:
        """Flatten ES _source to canonical schema."""
        src = hit.get("_source", {})
        return SearchResult(
            id=hit.get("_id") or src.get("id"),
            score=hit.get("_score", 0.0),
            rank=hit.get("sort", [0])[0] if "sort" in hit else 0,
            issuing_organ=src.get("issuing_organ"),
            body_plain=src.get("body_plain"),
            body_snippet=src.get("body_snippet"),
            title=src.get("title"),
            url=src.get("url"),
            date=src.get("date"),
        )
    
    def _normalize_pg_result(self, row: dict) -> SearchResult:
        """Ensure PG row dict matches canonical schema."""
        return SearchResult(
            id=row.get("id"),
            score=row.get("score", 0.0),
            rank=row.get("rank", 0),
            issuing_organ=row.get("issuing_organ"),
            body_plain=row.get("body_plain"),
            body_snippet=row.get("body_snippet"),
            title=row.get("title"),
            url=row.get("url"),
            date=row.get("date"),
        )
```

```python
SEARCH_SQL = """
SELECT 
    id, 
    issuing_organ, 
    COALESCE(LEFT(body_plain, 1000), LEFT(content, 1000)) AS body_snippet,
    body_plain,
    content AS body_plain_aliased,  -- Use content for grading consistency
    title,
    url,
    date,
    ts_rank(to_tsvector('portuguese', content), plainto_tsquery('portuguese', $1)) AS score
FROM documents d
LEFT JOIN LATERAL (
    SELECT name AS issuing_organ 
    FROM organizations 
    WHERE id = d.organ_id
    LIMIT 1
) org ON true
WHERE to_tsvector('portuguese', content) @@ plainto_tsquery('portuguese', $1)
ORDER BY score DESC
LIMIT $2
"""

# If 1:N detected, use DISTINCT ON:
SEARCH_SQL_DISTINCT = """
SELECT DISTINCT ON (d.id)
    d.id, 
    org.name AS issuing_organ, 
    LEFT(COALESCE(d.body_plain, d.content), 1000) AS body_snippet,
    d.body_plain,
    d.content AS body_plain_aliased,
    d.title,
    d.url,
    d.date,
    ts_rank(to_tsvector('portuguese', d.content), plainto_tsquery('portuguese', $1)) AS score
FROM documents d
LEFT JOIN organizations org ON d.organ_id = org.id
WHERE to_tsvector('portuguese', d.content) @@ plainto_tsquery('portuguese', $1)
ORDER BY d.id, score DESC
LIMIT $2
"""
```

**Cardinality Protection:**
- Use LATERAL subquery with LIMIT 1 (semantically equivalent to 1:1)
- If LATERAL not available, use DISTINCT ON (d.id)
- Add post-query assertion: `assert len(results) <= k` to detect multiplication

### 4.3 HybridAdapter Fix

**File: `src/adapters/hybrid_adapter.py`**

```python
def search(self, query: str, k: int = 100) -> list[SearchResult]:
    """
    Hybrid merges PG and ES results with field completeness guarantee.
    Strategy: Inner-join candidate IDs against both sources before merge.
    """
    # Step 1: Get candidates from both backends
    pg_results = self.pg_adapter.search(query, k=k*2)
    es_results = self.es_adapter.search(query, k=k*2)
    
    # Step 2: Build ID lookup with field completeness
    pg_by_id = {r.id: r for r in pg_results}
    es_by_id = {r.id: r for r in es_results}
    
    # Step 3: Inner-join to ensure field completeness
    # Use whichever source has non-null value for each field
    merged_ids = set(pg_by_id.keys()) | set(es_by_id.keys())
    final_results = []
    
    for doc_id in merged_ids:
        pg_doc = pg_by_id.get(doc_id)
        es_doc = es_by_id.get(doc_id)
        
        # Merge with field-level fallback
        merged = SearchResult(
            id=doc_id,
            score=(pg_doc.score if pg_doc else es_doc.score) if pg_doc or es_doc else 0.0,
            rank=0,  # Will be recalculated
            issuing_organ=pg_doc.issuing_organ if pg_doc and pg_doc.issuing_organ else (es_doc.issuing_organ if es_doc else None),
            body_plain=pg_doc.body_plain if pg_doc and pg_doc.body_plain else (es_doc.body_plain if es_doc else None),
            body_snippet=pg_doc.body_snippet if pg_doc and pg_doc.body_snippet else (es_doc.body_snippet if es_doc else None),
            title=pg_doc.title if pg_doc and pg_doc.title else (es_doc.title if es_doc else None),
            url=pg_doc.url if pg_doc and pg_doc.url else (es_doc.url if es_doc else None),
            date=pg_doc.date if pg_doc and pg_doc.date else (es_doc.date if es_doc else None),
        )
        final_results.append(merged)
    
    # Step 4: Re-rank by score and assign ranks
    final_results.sort(key=lambda x: x.score, reverse=True)
    for i, doc in enumerate(final_results[:k]):
        doc.rank = i + 1
    
    return final_results[:k]
```

## 5. Validation and Null Parity Layer

### 5.1 Optimized Validator (Replaces JSONSchema)

```python
# src/benchmark/validators.py
from dataclasses import fields
from typing import Any

class BenchmarkError(Exception):
    pass

REQUIRED_FIELDS = {"id", "score", "rank"}
OPTIONAL_FIELDS = {"issuing_organ", "body_plain", "body_snippet", "title", "url", "date", "metadata"}

def validate_results(results: list[dict], backend: str, debug: bool = False) -> None:
    """
    Fast runtime validation using set operations.
    Only runs full validation in debug mode or 1% sampling.
    """
    if not results:
        raise BenchmarkError(f"Backend {backend} returned empty results")
    
    # Fast path: validate first result only (sampling)
    if not debug and len(results) > 10:
        results = results[:1]
    
    for i, doc in enumerate(results):
        doc_keys = set(doc.keys())
        missing = REQUIRED_FIELDS - doc_keys
        if missing:
            raise BenchmarkError(
                f"Backend {backend} missing required keys at index {i}: {missing}"
            )
        # Warn about extra keys but don't fail
        extra = doc_keys - REQUIRED_FIELDS - OPTIONAL_FIELDS
        if extra and debug:
            print(f"Warning: backend {backend} has extra keys: {extra}")

def compute_null_frequency(results: list[dict], field: str) -> float:
    """Compute % of results where field is null or missing."""
    if not results:
        return 0.0
    null_count = sum(1 for r in results if r.get(field) is None or field not in r)
    return null_count / len(results)

def assert_null_parity(
    backends_results: dict[str, list[dict]], 
    fields: list[str],
    threshold: float = 0.10
) -> dict[str, dict[str, float]]:
    """
    Assert null frequency disparity between backends is within threshold.
    Returns null_frequency map for analysis.
    """
    null_freq = {}
    for backend, results in backends_results.items():
        null_freq[backend] = {}
        for field in fields:
            null_freq[backend][field] = compute_null_frequency(results, field)
    
    # Check disparity
    for field in fields:
        frequencies = [null_freq[b].get(field, 0) for b in backends_results]
        max_freq = max(frequencies)
        min_freq = min(frequencies)
        if max_freq - min_freq > threshold:
            raise BenchmarkError(
                f"Null frequency disparity for '{field}' exceeds {threshold}: "
                f"PG={null_freq.get('pg', {}).get(field):.1%}, "
                f"ES={null_freq.get('es', {}).get(field):.1%}, "
                f"Hybrid={null_freq.get('hybrid', {}).get(field):.1%}. "
                f"Either fix data source or adjust grader to exclude this field."
            )
    
    return null_freq
```

## 7. Severity-Ranked Concerns

| Severity | Concern | Mitigation |
|----------|---------|------------|
| **Critical** | Post-retrieval hydration masks production defect | BANNED. Fix at adapter level only |
| **Critical** | Benchmark runs with schema violations | Validation layer fails fast |
| **Critical** | 1:N JOIN multiplies rows silently | Use LATERAL LIMIT 1 or DISTINCT ON; post-query assertion |
| **Critical** | TOAST I/O dominates benchmark time | Use body_snippet; add covering index |
| **High** | Null frequency disparity bias (>10%) | assert_null_parity; grader adjustment required |
| **High** | ES _source vs PG row key path mismatch | Normalization layer in base adapter |
| **High** | Hybrid merge propagates nulls | Inner-join candidate IDs; field-level fallback |
| **High** | content ≠ body_plain causes false negatives | Use content for tsvector; alias as body_plain |
| **Medium** | JSONSchema overhead in hot path | Dataclass validation; sampling in production |
| **Medium** | Grader treats null as "no match" | Document grader behavior; add acceptance criterion |
| **Medium** | No rollback plan | Document rollback in docs/rollback_pg_adapter.md |
| **Low** | Validation misses data quality | Null frequency analysis per field per backend |

```python
# tests/test_adapter_schema.py
def test_pg_adapter_returns_required_fields():
    results = pg_adapter.search("test query")
    for doc in results:
        assert "id" in doc
        assert "issuing_organ" in doc
        assert "body_plain" in doc or "body_snippet" in doc
        assert "score" in doc

def test_pg_adapter_no_row_multiplication():
    results = pg_adapter.search("test query", k=50)
    assert len(results) <= 50, "JOIN multiplied rows!"

def test_null_values_permitted():
    results = [{"id": "1", "score": 0.5, "rank": 1, "issuing_organ": None}]
    validate_results(results, "pg")  # Should pass

def test_null_parity_threshold():
    pg_results = [{"id": "1", "score": 0.5, "rank": 1, "issuing_organ": None}] * 10
    es_results = [{"id": "1", "score": 0.5, "rank": 1, "issuing_organ": "org"}] * 10
    # 100% vs 0% = 100% disparity - should fail
    with raises(BenchmarkError, match="disparity"):
        assert_null_parity({"pg": pg_results, "es": es_results}, ["issuing_organ"], threshold=0.10)

def test_es_normalization():
    es_hit = {"_id": "1", "_score": 0.9, "_source": {"id": "1", "issuing_organ": "org"}}
    result = BaseAdapter()._normalize_es_result(es_hit)
    assert result.id == "1"
    assert result.issuing_organ == "org"

def test_hybrid_field_completeness():
    # Simulate hybrid merge with null fields
    hybrid = HybridAdapter(pg_adapter, es_adapter)
    results = hybrid.search("query", k=10)
    # Every result must have either PG or ES source for each field
    for doc in results:
        assert doc.id is not None

def test_content_body_plain_consistency():
    # Verify content used for search is what grader evaluates
    row = pg_adapter._normalize_pg_result({"id": "1", "content": "foo bar", "body_plain": "foo"})
    # If content != body_plain, grader should use content
    assert row.body_plain_aliased == "foo bar" or row.body_plain == "foo"
```
